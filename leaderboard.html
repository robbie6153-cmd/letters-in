import { db } from "./firebase-config.js"; import { collection, addDoc, serverTimestamp } from "https://www.gstatic.com/firebasejs/12.12.1/firebase-firestore.js"; const GAME_ID = "lettersin"; window.submitRobTechScore = async function(score) { if (!window.currentRobTechUser) { alert("You must create a RobTechUK Games account to submit your score to the leaderboard."); return; } await addDoc(collection(db, "leaderboards_lettersin"), { uid: window.currentRobTechUser.uid, username: window.currentRobTechUser.username, score: Number(score), game: GAME_ID, createdAt: serverTimestamp() }); window.location.href = "leaderboard.html"; };
